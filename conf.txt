<VirtualHost *:80>
    ServerAdmin admin@chainai.online
    ServerName chainai.online
    ServerAlias www.chainai.online

    DocumentRoot /var/www/django/ubuntu-deployment/


    ErrorLog ${APACHE_LOG_DIR}/chainai.online_error.log
    CustomLog ${APACHE_LOG_DIR}/chainai.online_access.log combined

    Alias /static /var/www/django/ubuntu-deployment/static
    <Directory /var/www/django/ubuntu-deployment/static>
            Require all granted
    </Directory>

    Alias /media /var/www/django/ubuntu-deployment/media
    <Directory /var/www/django/ubuntu-deployment/media>
            Require all granted
     </Directory>


    Alias /staticfiles /var/www/django/ubuntu-deployment/staticfiles
    <Directory /var/www/django/ubuntu-deployment/staticfiles>
            Require all granted
     </Directory>

    <Directory /var/www/django/ubuntu-deployment/core>
            <Files wsgi.py>
                    Require all granted
            </Files>
    </Directory>

    WSGIDaemonProcess django_app python-path=/var/www/django python-home=/var/www/django/environment
    WSGIProcessGroup django_app
    WSGIScriptAlias / /var/www/django/ubuntu-deployment/core/wsgi.py
</VirtualHost>
